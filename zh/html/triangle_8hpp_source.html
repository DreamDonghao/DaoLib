9 /// @brief 三角形几何图元
10 /// @details 表示一个由三个顶点组成的三角形，支持设置位置、颜色等属性
11 /// 提供按索引和按顶点名称（A、B、C）两种访问方式，便于灵活操作
12 class Triangle {
13 public:
14 /// @brief 顶点索引枚举
15 enum VertexIndex : size_t {

21 /// @brief 默认构造函数，创建白色三角形（三个顶点均在原点）
22 Triangle() : m_vertices{Vertex{}, Vertex{}, Vertex{}} {
23 }

28 /// @param vertexC 顶点C

31 }

33 /// @brief 通过索引设置顶点属性（位置和颜色）

52 /// @param y 新的y坐标

67 [[nodiscard]] const Vertex &vertex(const size_t index) const {
68 return m_vertices[index];
69 }

72 /// @param color 新的颜色

75 vertex.setColor(color);

80 /// @param dx x方向平移量
81 /// @param dy y方向平移量

88 /// @brief 检查三角形是否有效（三个顶点不共线）
89 /// @note 当前为简单实现，仅检查顶点是否不同
90 /// @return 如果三个顶点不同返回true，否则false
91 [[nodiscard]] bool isValid() const {
92 const auto &v0 = m_vertices[0];
93 const auto &v1 = m_vertices[1];
94 const auto &v2 = m_vertices[2];
95
96 // 简单检查：三个顶点不应完全相同
97 return !(v0.x() == v1.x() && v0.y() == v1.y() &&
98 v1.x() == v2.x() && v1.y() == v2.y());
99 }
