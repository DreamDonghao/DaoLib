12 /// @brief 托盘回调函数适配器
13 /// @details 将 SDL 的 C 风格回调转换为 C++ 函数对象调用
14 /// @param userdata 用户数据（TrayFunc 指针）

16 inline void callback(void *userdata, SDL_TrayEntry *entry) {
17 (*static_cast<TrayFunc *>(userdata))(entry);

20 /// @brief 托盘

24 /// @param iconPath 托盘图标文件路径

26 Tray(const std::string_view iconPath, const std::string_view tooltip) {
27 m_icon = IMG_Load(iconPath.data());
28 if (!m_icon) {
29 DAO_ERROR_LOG(std::string("托盘图片失败:") + iconPath.data());
30 }
31 m_tray = SDL_CreateTray(m_icon, tooltip.data());
32 if (!m_tray) {
33 DAO_ERROR_LOG("托盘创建失败");
34 }

36 if (!m_menu) {
37 DAO_ERROR_LOG("托盘菜单创建失败");
38 }
39 }

42 /// @param label 菜单项显示文本
43 /// @param trayFunc 回调函数引用
44 /// @warning 使用此版本时需确保 trayFunc 生命周期长于托盘对象
45 void button(const std::string_view label, TrayFunc &trayFunc) const {

48 }

55 SDL_TrayEntry *entry = SDL_InsertTrayEntryAt(m_menu, -1, label.data(), SDL_TRAYENTRY_BUTTON);

57 }
