17 /// @brief 判断当前是否到达触发时间
18 ///
19 /// 行为：
20 /// - 到达间隔 → 返回 true，并推进内部时间
21 /// - 未到 → 返回 false
22 ///
23 /// 特别注意：
24 /// 如果主线程卡顿较久（例如 40ms，而 interval=16ms），

26 /// 从而保证时间轴连续，不丢逻辑帧。
27 explicit operator bool() {
28 if (const auto now = clock::now(); now - m_last >= m_interval) {
29 // 防止时间漂移：按固定步长推进，而不是直接跳到 now
30 m_last += m_interval;
31 return true;
32 }
33 return false;
34 }

38 m_last = clock::now();
39 }

43 m_interval = interval;
44 }
45
