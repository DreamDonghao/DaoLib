7void dao::WindowController::executeCommand(Window &window) {
8 for (const auto &[type, data]: m_windowCommands) {
9 u32 x, y;
10 switch (type) {
11 case WindowCommandType::switchPage:
12 window.switchPage(std::any_cast<std::string>(data));
13 break;
14 case WindowCommandType::setClickThrough:
15 window.setClickThrough(std::any_cast<bool>(data));
16 break;
17 case WindowCommandType::setPosition:
18 x = std::any_cast<std::pair<u32, u32> >(data).first;
19 y = std::any_cast<std::pair<u32, u32> >(data).second;
20 window.setPosition(x, y);
21 break;
22 case WindowCommandType::movePosition:
23 x = std::any_cast<std::pair<u32, u32> >(data).first;
24 y = std::any_cast<std::pair<u32, u32> >(data).second;
25 window.movePosition(x, y);
26 break;
27 case WindowCommandType::setSize:
28 x = std::any_cast<std::pair<u32, u32> >(data).first;
29 y = std::any_cast<std::pair<u32, u32> >(data).second;
30 window.setSize(x, y);
31 default:
32 break;
33 }
34 }
35 m_windowCommands.clear();
36}
