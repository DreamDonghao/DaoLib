41 /// @brief 加载字形
42 /// @details 将一个文字的字形编码添加到字形图集中
43 /// @param charCode 文字的utf-32编码
44 void registerGlyph(const utf32char charCode) {
45 if (m_glyphs.contains(charCode)) { return; }
46
47 m_isUpdated = true;
48 SDL_Surface *glyphSurface = TTF_RenderGlyph_Blended(m_font, charCode, SDL_Color{255, 255, 255, 255});
49 m_cursor.rowHeight = std::max(m_cursor.rowHeight, glyphSurface->h);
50 m_glyphAspectRatios.emplace(charCode, ratio(glyphSurface->w, m_cursor.rowHeight));
51
52 if (m_cursor.x + glyphSurface->w > m_atlasSize) {
53 m_cursor.x = 0;
54 m_cursor.y += m_cursor.rowHeight;
55 m_cursor.rowHeight = 0;
56 }
57 const SDL_Rect dst{m_cursor.x, m_cursor.y, glyphSurface->w, glyphSurface->h};
58 SDL_BlitSurface(glyphSurface, nullptr, m_atlasSurface, &dst);
59 int minx, maxx, miny, maxy, advance;
60 TTF_GetGlyphMetrics(m_font, charCode, &minx, &maxx, &miny, &maxy, &advance);
61
62 Glyph glyph{};
63 glyph.size = {static_cast<float>(glyphSurface->w), static_cast<float>(glyphSurface->h)};
64 glyph.bearing = {static_cast<float>(minx), static_cast<float>(maxy)};
65 glyph.advance = static_cast<float>(advance);
66
67 glyph.pos = {
68 ratio(dst.x, m_atlasSize), ratio(dst.y, m_atlasSize),
69 ratio(dst.w, m_atlasSize), ratio(dst.h, m_atlasSize)
70 };
71
72 m_glyphs.emplace(charCode, glyph);
73 m_cursor.x += glyphSurface->w;
74
75 SDL_DestroySurface(glyphSurface);
76 }
