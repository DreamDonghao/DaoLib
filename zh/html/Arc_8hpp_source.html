110 /// @brief 更新顶点位置
111 void updateVertices(const ColorRGBA &color) {
112 // 每段需要4个顶点（内外各2个）
113 m_vertices.resize((m_segments + 1) * 2);
114
115 const f32 angleStep = (m_endAngle - m_startAngle) / static_cast<f32>(m_segments);
116 const f32 innerR = m_radius - m_thickness / 2.0f;
117 const f32 outerR = m_radius + m_thickness / 2.0f;
118
119 for (i32 i = 0; i <= m_segments; ++i) {
120 const f32 angle = m_startAngle + i * angleStep;
121 const f32 cosA = std::cos(angle);
122 const f32 sinA = std::sin(angle);
123
124 m_vertices[i * 2] = Vertex(
125 m_cx + innerR * cosA,
126 m_cy + innerR * sinA,
127 color);
128 m_vertices[i * 2 + 1] = Vertex(
129 m_cx + outerR * cosA,
130 m_cy + outerR * sinA,
131 color);
132 }
133 }
