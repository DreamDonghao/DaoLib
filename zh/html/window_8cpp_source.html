40 Window &Window::addPage(std::unique_ptr<Page> &&page) {
41 const std::string title = page->getTitle();
42 detectionError(!m_pages.contains(title), std::string("重复页面:") + title);
43 for (auto textureId: page->getRegisterTexture()) {
44 registerTexture(textureId);
45 }
46 if (m_pages.empty()) {
47 m_nowPageTitle = title;
48 }
49 m_pages[title] = std::move(page);
50 m_pages[title]->init();
51 m_atlasTextures[1] = SDL_CreateTextureFromSurface(
52 m_renderer, &m_pages[title]->getGlyphAtlas().getAtlasSurface());
53 return *this;
54 }

56 void Window::registerTexture(const uint32 &textureId) {
57 const AtlasRegion atlasRegion = getAtlasRegion(textureId);
58 // 加载新纹理图集
59 if (const uint32 atlasId = atlasRegion.atlasId; !m_atlasTextures.contains(atlasId)) {
60 const char *texturePath = atlasRegion.atlasPath;
61 m_atlasTextures[atlasId] = IMG_LoadTexture(m_renderer, texturePath);
62 SDL_SetTextureScaleMode(m_atlasTextures[atlasId], SDL_SCALEMODE_NEAREST);
63 detectionError(m_atlasTextures[atlasId],
64 std::string("纹理图集加载失败") + SDL_GetError() + ":" + texturePath);
65 }
66 }

68 void Window::update() {
69 m_pages[m_nowPageTitle]->update();
70 if (m_pages[m_nowPageTitle]->getGlyphAtlas().isUpdated()) {
71 SDL_UpdateTexture(
72 m_atlasTextures[1], nullptr,
73 m_pages[m_nowPageTitle]->getGlyphAtlas().getAtlasSurface().pixels,
74 m_pages[m_nowPageTitle]->getGlyphAtlas().getAtlasSurface().pitch
75 );
76 SDL_SetTextureScaleMode(m_atlasTextures[1], SDL_SCALEMODE_NEAREST);
77 m_pages[m_nowPageTitle]->getGlyphAtlas().clearUpdateFlag();
78 }
79
80 if (m_pages[m_nowPageTitle]->getEvent().isPresenceMessage()) {
81 switch (const auto event = m_pages[m_nowPageTitle]->getEvent().popEvent(); event.type) {
82 case PageCmdType::switchPage:
83 switchPage(std::any_cast<std::string>(event.data));
84 break;
85 default:
86 break;
87 }
88 }

103 void Window::render() {
104 SDL_RenderClear(m_renderer);
105 for (const auto &[atlasId, vertices, indices,indicesCount]:
106 m_pages[m_nowPageTitle]->getDrawBatches()) {
107 SDL_RenderGeometry(
108 m_renderer, m_atlasTextures[atlasId],
109 vertices.data(), static_cast<int>(vertices.size()),
110 indices->data(), indicesCount
111 );
112 }
113
114 SDL_RenderPresent(m_renderer);
115 }
