11 constexpr std::array<float, 256> rgbNorm = [] {
12 std::array<float, 256> lut{};

19 /// @brief RGBA 颜色
20 struct ColorRGBA {
21 uint32_t r, g, b;
22 float a;
23 };

25 constexpr uint8 hexByteToU8(const std::string_view sv) noexcept {
26 uint8 value = 0;
27 for (const char c: sv) {
28 value *= 16;
29 if (c >= '0' && c <= '9') value += c - '0';
30 else if (c >= 'A' && c <= 'F') value += c - 'A' + 10;
31 else if (c >= 'a' && c <= 'f') value += c - 'a' + 10;
32 else return 0;
33 }
34 return value;
35 }

37 /// @brief 将 hex 颜色格式转换为 RGBA 格式
38 constexpr ColorRGBA hexToRGBA(std::string_view hex) {
39 if (!hex.empty() && hex[0] == '#')
40 hex.remove_prefix(1);
41 const auto len = hex.size();
42 if (len != 6 && len != 8) {
43 return {255, 255, 255, 1.0f};
44 }
45 const uint8 r = hexByteToU8(hex.substr(0, 2));
46 const uint8 g = hexByteToU8(hex.substr(2, 2));
47 const uint8 b = hexByteToU8(hex.substr(4, 2));
48 const float32 a = (len == 8)? rgbNorm[hexByteToU8(hex.substr(6, 2))]: 1.0f;
49
50 if ((r == 0 && (hex[0] != '0' || hex[1] != '0')) ||
51 (g == 0 && (hex[2] != '0' || hex[3] != '0')) ||
52 (b == 0 && (hex[4] != '0' || hex[5] != '0')) ||
53 (len == 8 && hexByteToU8(hex.substr(6, 2)) == 0 && (hex[6] != '0' || hex[7] != '0'))) {
54 return {255, 255, 255, 1.0f};
55 }
56 return {r, g, b, a};
57 }
58}
