7void dao::WindowController::executeCommand(Window &window) {
8 for (const auto &[type, data]: m_windowCommands) {
9 uint32 x, y;
10 switch (type) {
11 case WindowCommandType::switchPage:
12 window.switchPage(std::any_cast<std::string>(data));
13 break;
14 case WindowCommandType::setClickThrough:
15 window.setClickThrough(std::any_cast<bool>(data));
16 break;
17 case WindowCommandType::setPosition:
18 x = std::any_cast<std::pair<uint32, uint32> >(data).first;
19 y = std::any_cast<std::pair<uint32, uint32> >(data).second;
20 window.setPosition(x, y);
21 break;
22 case WindowCommandType::setSize:
23 x = std::any_cast<std::pair<uint32, uint32> >(data).first;
24 y = std::any_cast<std::pair<uint32, uint32> >(data).second;
25 window.setSize(x, y);
26 default:
27 break;
28 }
29 }
30 m_windowCommands.clear();
31}
