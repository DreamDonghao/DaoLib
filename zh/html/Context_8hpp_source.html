10 /// @brief 上下文容器类
11 /// @details 用于存储和检索任意类型的共享对象，通过类型信息作为键进行索引。
12 /// 支持两种模式：
13 /// - 外部对象引用：通过 emplace 引用已存在的对象，不持有所有权
14 /// - 内部对象创建：在容器内创建新对象，容器持有所有权
15 class Context {
16 public:
17 Context() = default;
18
19 /// @brief 向容器中插入或创建对象
20 /// @tparam T 对象类型
21 /// @tparam Args 构造参数类型
22 /// @param args 若为单个 T 类型参数，则引用该对象；否则用于构造新对象
23 /// @returns 对象的引用
24 /// @note 当传入单个 T 类型参数时，仅持有引用而不获取所有权；
25 /// 当传入其他参数时，容器创建并拥有新对象
26 template<typename T, typename... Args>
27 T &emplace(Args &&...args) {
28 if constexpr (sizeof...(args) == 1 &&
29 std::is_same_v<T, std::decay_t<std::tuple_element_t<0, std::tuple<Args...>>>>) {
30 // 引用模式：传入单个 T 类型对象，不持有所有权
31 auto &context = std::get<0>(std::forward_as_tuple(std::forward<Args>(args)...));
32 m_context[typeid(T)] = std::shared_ptr<void>(std::addressof(context), [](void *) {});
33 return context;
34 } else {
35 // 创建模式：构造新对象并持有所有权
36 auto ptr = std::make_shared<T>(std::forward<Args>(args)...);
37 m_context[typeid(T)] = ptr;
38 return *ptr;
39 }
40 }
41
42 /// @brief 获取容器中的对象
43 /// @tparam T 对象类型
44 /// @returns 对象指针，若不存在则返回 nullptr
45 template<typename T>
46 T *get() {
47 const auto it = m_context.find(typeid(T));
48 if (it == m_context.end()) {
49 DAO_ERROR_LOG("状态不存在");
50 return nullptr;
51 }
52 return std::static_pointer_cast<T>(it->second).get();
53 }
54
55 /// @brief 检查容器中是否存在指定类型的对象
56 /// @tparam T 对象类型
57 /// @returns 存在返回 true，否则返回 false
58 template<typename T>
59 [[nodiscard]] bool has() const {
60 return m_context.contains(typeid(T));
61 }
62
63 private:
64 hash_map<std::type_index, std::shared_ptr<void>> m_context; ///< 类型索引到对象的映射表
65 };
66} // namespace dao
