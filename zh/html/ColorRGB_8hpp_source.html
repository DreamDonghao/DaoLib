115 /// @brief 将十六进制颜色字符串转换为ColorRGBA对象
116 /// @param hex 十六进制颜色字符串，支持格式：RRGGBB、#RRGGBB、RRGGBBAA或#RRGGBBAA
117 /// @return 转换后的ColorRGBA对象，如果格式无效则返回白色(#FFFFFF)
118 /// @details 算法步骤：
119 /// 1. 移除可选的'#'前缀
120 /// 2. 验证长度必须是6(RGB)或8(RGBA)
121 /// 3. 将每两个字符转换为一个字节值(0-255)
122 /// 4. 检测转换错误：值为0但对应的十六进制字符不是"00"
123 static constexpr ColorRGBA hexToRGBA(std::string_view hex) noexcept {
124 // 移除'#'前缀
125 if (!hex.empty() && hex[0] == '#')
126 hex.remove_prefix(1);
127
128 const auto len = hex.size();
129 // 验证长度
130 if (len != 6 && len != 8) {
131 return ColorRGBA{255, 255, 255, 1.0f}; // 无效格式，返回白色
132 }
133
134 // 转换RGB分量
135 const i32 r = hexByteToU8(hex.substr(0, 2));
136 const i32 g = hexByteToU8(hex.substr(2, 2));
137 const i32 b = hexByteToU8(hex.substr(4, 2));
138
139 // 转换Alpha分量（如果存在）
140 const f32 a = (len == 8) ? rgbNorm[hexByteToU8(hex.substr(6, 2))] : 1.0f;
141
142 // 验证转换结果：检查是否因无效字符导致转换失败（返回0但hex不是"00"）
143 const auto isValidHexByte = [&hex](i32 value, size_t pos1, size_t pos2) {
144 return !(value == 0 && (hex[pos1] != '0' || hex[pos2] != '0'));
145 };
146
147 if (!isValidHexByte(r, 0, 1) || !isValidHexByte(g, 2, 3) || !isValidHexByte(b, 4, 5) ||
148 (len == 8 && !isValidHexByte(hexByteToU8(hex.substr(6, 2)), 6, 7))) {
149 return ColorRGBA{255, 255, 255, 1.0f}; // 无效字符，返回白色
150 }
151
152 return ColorRGBA{r, g, b, a};
153 }
154
155 explicit constexpr operator SDL_FColor() const noexcept {
156 return {rgbNorm[m_r], rgbNorm[m_g], rgbNorm[m_b], m_a};
157 }

163 f32 m_a; ///< 透明度值 (0.0-1.0)

165 /// @brief 将两个十六进制字符转换为对应的字节值(0-255)
166 /// @param sv 包含1-2个十六进制字符的字符串视图
167 /// @return 转换后的字节值，如果包含无效字符则返回0
168 /// @details 支持大写和小写十六进制字符(A-F, a-f)。算法从左到右处理每个字符，
169 /// 每次将当前值乘以16并加上新字符的值。
170 static constexpr i32 hexByteToU8(const std::string_view sv) noexcept {
171 i32 value = 0;
172 for (const char c: sv) {
173 value *= 16;
174 if (c >= '0' && c <= '9') {
175 value += static_cast<i32>(c - '0');
176 } else if (c >= 'A' && c <= 'F') {
177 value += static_cast<i32>(c - 'A' + 10);
178 } else if (c >= 'a' && c <= 'f') {
179 value += static_cast<i32>(c - 'a' + 10);
180 } else {
181 return 0; // 无效字符
182 }
183 }
184 return value;
185 }
186 };
