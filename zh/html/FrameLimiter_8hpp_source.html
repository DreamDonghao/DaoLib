7 /// @brief 帧率限制器
8 /// @details 用于精确控制应用程序的帧率，确保帧率稳定在目标值
9 /// 采用混合等待策略：先睡眠等待大部分时间，然后自旋等待剩余时间
10 class FrameLimiter {
11 public:
12 /// @brief 构造函数
13 /// @param target_fps 目标帧率（每秒帧数）
14 explicit FrameLimiter(const i32 target_fps)
15 : m_frame_duration(std::chrono::nanoseconds(1'000'000'000 / target_fps)), // 计算每帧持续时间（纳秒）
16 m_last_frame_time(std::chrono::steady_clock::now()), // 记录上一帧的时间点
17 m_spin_threshold(std::chrono::milliseconds(2)) {
18 // 自旋等待阈值（2毫秒）
19 }
20
21 /// @brief 等待下一帧
22 /// @details 混合等待策略：
23 /// 1. 计算已用时间
24 /// 2. 如果剩余时间大于自旋阈值，先睡眠等待大部分时间
25 /// 3. 自旋等待剩余时间，确保精确到达下一帧时间点
26 void wait() {
27 const auto now = std::chrono::steady_clock::now(); // 获取当前时间
28 const auto elapsed = now - m_last_frame_time; // 计算自上一帧以来的时间
29
30 // 如果剩余时间大于自旋阈值，先进行睡眠等待
31 if (const auto sleep_duration = m_frame_duration - elapsed; sleep_duration > m_spin_threshold) {
32 const auto sleep_time = sleep_duration - m_spin_threshold; // 睡眠时间 = 剩余时间 - 自旋阈值
33 std::this_thread::sleep_for(sleep_time); // 睡眠等待
34 }
35
36 // 自旋等待直到达到目标帧时间
37 while (std::chrono::steady_clock::now() - m_last_frame_time < m_frame_duration) {
38 std::this_thread::yield(); // 让出CPU时间片
39 }
40
41 m_last_frame_time = std::chrono::steady_clock::now(); // 更新上一帧时间
42 }
43
44 /// @brief 动态设置目标帧率
45 /// @param target_fps 新的目标帧率
46 void setFPS(const i32 target_fps) {
47 m_frame_duration = std::chrono::nanoseconds(1'000'000'000 / target_fps); // 重新计算每帧持续时间
48 }
49
50 private:
51 std::chrono::nanoseconds m_frame_duration; ///< 每帧的目标持续时间（纳秒）
52 std::chrono::steady_clock::time_point m_last_frame_time; ///< 上一帧的时间点
53 std::chrono::nanoseconds m_spin_threshold; ///< 自旋等待阈值（2毫秒）
54 };
