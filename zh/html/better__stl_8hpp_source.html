54 inline utf32str utf8ToUtf32(const std::string &s) {
55 std::u32string out;
56 out.reserve(s.size());
57
58 size_t i = 0;
59 const size_t size = s.size();
60
61 while (i < size) {
62 const unsigned char c = static_cast<unsigned char>(s[i]);
63 u32 codepoint = 0;
64 size_t len = 0;
65 if (c < 0x80) {
66 codepoint = c;
67 len = 1;
68 } else if ((c & 0xE0) == 0xC0) {
69 if (i + 1 >= size) break;
70 codepoint = (c & 0x1F) << 6 |
71 static_cast<unsigned char>(s[i + 1]) & 0x3F;
72 len = 2;
73 } else if ((c & 0xF0) == 0xE0) {
74 if (i + 2 >= size) break;
75 codepoint =
76 (c & 0x0F) << 12 |
77 (static_cast<unsigned char>(s[i + 1]) & 0x3F) << 6 |
78 static_cast<unsigned char>(s[i + 2]) & 0x3F;
79 len = 3;
80 } else if ((c & 0xF8) == 0xF0) {
81 if (i + 3 >= size) break;
82 codepoint = (c & 0x07) << 18 |
83 (static_cast<unsigned char>(s[i + 1]) & 0x3F) << 12 |
84 (static_cast<unsigned char>(s[i + 2]) & 0x3F) << 6 |
85 static_cast<unsigned char>(s[i + 3]) & 0x3F;
86 len = 4;
87 } else {
88 codepoint = 0xFFFD;
89 len = 1;
90 }
91 out.push_back(codepoint);
92 i += len;
93 }
94
95 return out;
96 }
