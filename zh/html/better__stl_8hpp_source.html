81 inline std::u32string utf8ToUtf32(const std::string &utf8Str) {
82 std::u32string utf32Str;
83
84 for (size_t i = 0; i < utf8Str.size();) {
85 uint32_t codepoint = 0;
86
87 if (const unsigned char c = utf8Str[i]; c < 0x80) {
88 // 单字节 UTF-8
89 codepoint = c;
90 i += 1;
91 } else if ((c & 0xE0) == 0xC0) {
92 // 双字节 UTF-8
93 codepoint = (c & 0x1F) << 6 | utf8Str[i + 1] & 0x3F;
94 i += 2;
95 } else if ((c & 0xF0) == 0xE0) {
96 // 三字节 UTF-8
97 codepoint = ((c & 0x0F) << 12) | ((utf8Str[i + 1] & 0x3F) << 6) | (utf8Str[i + 2] & 0x3F);
98 i += 3;
99 } else if ((c & 0xF8) == 0xF0) {
100 // 四字节 UTF-8
101 codepoint = ((c & 0x07) << 18) | ((utf8Str[i + 1] & 0x3F) << 12) |
102 ((utf8Str[i + 2] & 0x3F) << 6) | (utf8Str[i + 3] & 0x3F);
103 i += 4;
104 }
105
106 utf32Str.push_back(codepoint);
107 }
108
109 return utf32Str;
110 }
