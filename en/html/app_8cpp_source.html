29 void App::run() {
30 m_running = true;
31 while (m_running) {
32 for (const auto &window: m_windows | std::views::values) {
33 window->update();
34 window->render();
35 window->getAppController().executeCommand(*this);
36 }
37 SDL_Event event;
38 while (SDL_PollEvent(&event)) {
39 if (event.type == SDL_EVENT_WINDOW_CLOSE_REQUESTED) {
40 uint32_t wid = event.window.windowID;
41 if (auto it = m_windows.find(wid); it != m_windows.end()) {
42 it->second->hide();
43 }
44 }
45
46 uint32_t wid = 0;
47 if (event.type >= SDL_EVENT_WINDOW_FIRST && event.type <= SDL_EVENT_WINDOW_LAST) {
48 wid = event.window.windowID;
49 } else if (event.type >= SDL_EVENT_KEY_DOWN && event.type <= SDL_EVENT_KEY_UP) {
50 wid = event.key.windowID;
51 } else if (event.type >= SDL_EVENT_MOUSE_MOTION && event.type <= SDL_EVENT_MOUSE_WHEEL) {
52 wid = event.button.windowID;
53 }
54
55 if (wid != 0) {
56 if (auto it = m_windows.find(wid); it != m_windows.end()) {
57 it->second->handleInputEvent(event);
58 }
59 }
60 }
61 }
62 }
