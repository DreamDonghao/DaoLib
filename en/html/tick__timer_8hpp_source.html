10 /// @brief 节拍器
11 class TickTimer {
12 public:
13 using clock = std::chrono::steady_clock;
14
15 /// @param interval 触发间隔（如 16ms ≈ 60fps）
16 explicit TickTimer(const std::chrono::milliseconds interval)
17 : m_interval(interval),
18 m_last(clock::now()) {
19 }
20
21 /// @brief 判断当前是否到达触发时间
22 ///
23 /// 行为：
24 /// - 到达间隔 → 返回 true，并推进内部时间
25 /// - 未到 → 返回 false
26 ///
27 /// 特别注意：
28 /// 如果主线程卡顿较久（例如 40ms，而 interval=16ms），
29 /// 本函数会只推进一个 interval，下一次调用仍可能返回 true，
30 /// 从而保证时间轴连续，不丢逻辑帧。
31 explicit operator bool() {
32 if (const auto now = clock::now(); now - m_last >= m_interval) {
33 // 防止时间漂移：按固定步长推进，而不是直接跳到 now
34 m_last += m_interval;
35 return true;
36 }
37 return false;
38 }
39
40 /// @brief 重置计时器，从当前时间重新开始
41 void reset() {
42 m_last = clock::now();
43 }
44
45 /// @brief 修改触发间隔
46 void setInterval(std::chrono::milliseconds interval) {
47 m_interval = interval;
48 }
49
50 /// @brief 获取当前间隔
51 [[nodiscard]]
52 std::chrono::milliseconds interval() const {
53 return m_interval;
54 }
55
56 private:
57 std::chrono::milliseconds m_interval; ///< 触发间隔
58 clock::time_point m_last; ///< 上次触发时间点
59 };
