24 /// @brief 处理消息，更新状态
25 void handleEvent(const SDL_Event &event) {
26 if (m_status == InputStatus::Disabled) {
27 return;
28 }
29 float x = 0, y = 0;
30 if (event.type == SDL_EVENT_MOUSE_MOTION) {
31 x = event.motion.x;
32 y = event.motion.y;
33 } else if (event.type == SDL_EVENT_MOUSE_BUTTON_DOWN || event.type == SDL_EVENT_MOUSE_BUTTON_UP) {
34 x = event.button.x;
35 y = event.button.y;
36 }
37
38 SDL_Window *win = SDL_GetWindowFromID(event.window.windowID);
39 if (m_status == InputStatus::Input) {
40 const SDL_Rect rect{
41 static_cast<int>(m_boundingBox.getLeft()),
42 static_cast<int>(m_boundingBox.getTop()), // 顶部
43 static_cast<int>(m_boundingBox.getWidth()), // 宽度
44 static_cast<int>(m_boundingBox.getHeight()) // 高度
45 };
46 SDL_SetTextInputArea(win, &rect, cursorPos);
47 if (event.type == SDL_EVENT_TEXT_INPUT) {
48 const char *utf8chars = event.text.text;
49 const auto begin = utf8::unchecked::iterator(utf8chars);
50 const auto end = utf8::unchecked::iterator(utf8chars + std::strlen(utf8chars));
51
52 for (auto it = begin; it != end; ++it) {
53 m_text.insert(m_text.begin() + cursorPos, static_cast<char32_t>(*it));
54 cursorPos += 1;
55 }
56 } else if (event.type == SDL_EVENT_KEY_DOWN) {
57 if (event.key.key == SDLK_BACKSPACE && !m_text.empty()) {
58 if (cursorPos != 0) {
59 m_text.erase(cursorPos - 1, 1);
60 --cursorPos;
61 }
62 } else if (event.key.key == SDLK_DELETE) {
63 if (cursorPos != m_text.size()) {
64 m_text.erase(cursorPos, 1);
65 }
66 } else if (event.key.key == SDLK_LEFT) {
67 cursorPos = std::max(cursorPos - 1, 0);
68 } else if (event.key.key == SDLK_RIGHT) {
69 cursorPos = std::min(cursorPos + 1, static_cast<int>(m_text.size()));
70 }
71 } else if (event.type == SDL_EVENT_MOUSE_BUTTON_DOWN && event.button.button == SDL_BUTTON_LEFT) {
72 if (!m_boundingBox.isInBoundingBox(x, y)) {
73 m_status = InputStatus::Normal;
74 SDL_StopTextInput(win);
75 }
76 }
77 } else if (event.type == SDL_EVENT_MOUSE_BUTTON_DOWN && event.button.button == SDL_BUTTON_LEFT) {
78 if (m_boundingBox.isInBoundingBox(x, y)) {
79 m_status = InputStatus::Input;
80 SDL_StartTextInput(win);
81 }
82 }
83 }
