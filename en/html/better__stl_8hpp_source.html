79 inline utf32str utf8ToUtf32(const std::string &s) {
80 std::u32string out;
81 out.reserve(s.size());
82
83 size_t i = 0;
84 const size_t size = s.size();
85
86 while (i < size) {
87 const unsigned char c = static_cast<unsigned char>(s[i]);
88 u32 codepoint = 0;
89 size_t len = 0;
90 if (c < 0x80) {
91 codepoint = c;
92 len = 1;
93 } else if ((c & 0xE0) == 0xC0) {
94 if (i + 1 >= size) break;
95 codepoint = (c & 0x1F) << 6 |
96 static_cast<unsigned char>(s[i + 1]) & 0x3F;
97 len = 2;
98 } else if ((c & 0xF0) == 0xE0) {
99 if (i + 2 >= size) break;
100 codepoint =
101 (c & 0x0F) << 12 |
102 (static_cast<unsigned char>(s[i + 1]) & 0x3F) << 6 |
103 static_cast<unsigned char>(s[i + 2]) & 0x3F;
104 len = 3;
105 } else if ((c & 0xF8) == 0xF0) {
106 if (i + 3 >= size) break;
107 codepoint = (c & 0x07) << 18 |
108 (static_cast<unsigned char>(s[i + 1]) & 0x3F) << 12 |
109 (static_cast<unsigned char>(s[i + 2]) & 0x3F) << 6 |
110 static_cast<unsigned char>(s[i + 3]) & 0x3F;
111 len = 4;
112 } else {
113 codepoint = 0xFFFD;
114 len = 1;
115 }
116 out.push_back(codepoint);
117 i += len;
118 }
119
120 return out;
121 }
