78 // 扇形由 (segments) 个三角形组成

82 *vertices++ = m_vertices[i + 1].getSDLVertex(); // 当前边缘点

97 /// @brief 更新顶点位置
98 void updateVertices(const ColorRGBA &color) {
99 m_vertices.resize(m_segments + 2);
100 m_vertices[0] = Vertex(m_cx, m_cy, color); // 圆心
101
102 const f32 angleStep = (m_endAngle - m_startAngle) / static_cast<f32>(m_segments);
103 for (i32 i = 0; i <= m_segments; ++i) {
104 const f32 angle = m_startAngle + i * angleStep;
105 const f32 x = m_cx + m_radius * std::cos(angle);
106 const f32 y = m_cy + m_radius * std::sin(angle);
107 m_vertices[i + 1] = Vertex(x, y, color);
108 }
