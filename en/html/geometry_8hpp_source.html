18 /// @brief 到SDL_Vertex的类型转换
19 /// @details 纹理坐标用 1*1px 的纯白图片,将 255RGB 数据归一化
20 explicit operator SDL_Vertex() const {
21 SDL_Vertex result;
22 result.position = {x, y};
23 result.color = {rgbNorm[color.r], rgbNorm[color.g], rgbNorm[color.b], color.a};
24 result.tex_coord = {1.0f, 1.0f};

26 }

29 /// @brief 将 GeometryVertex 批量转换为 SDL_Vertex
30 /// @tparam N 顶点个数
31 template<size_t N>
32 static std::array<SDL_Vertex, N> makeVertices(const std::array<GeometryVertex, N> &v) {
33 std::array<SDL_Vertex, N> arr{};
34 for (size_t i = 0; i < N; ++i) {
35 arr[i] = static_cast<SDL_Vertex>(v[i]);
36 }
37 return arr;
38 }

40 /// @brief 几何图形
41 /// @details 一个多边形
42 /// @tparam N 多边形边数
43 template<size_t N>
44 class Geometry {
45 public:
46 explicit Geometry(const std::array<GeometryVertex, N> &vertices): m_vertices(makeVertices(vertices)) {
47 int index = -1;
48 for (i32 i = 1; i < N - 1; ++i) {
49 m_indices[++index] = 0;
50 m_indices[++index] = i;
51 m_indices[++index] = i + 1;
52 }
53 }
54
55 void moveXY(const f32 x, const f32 y) {
56 for (i32 i = 0; i < N; ++i) {
57 m_vertices[i].position.x += x;
58 m_vertices[i].position.y += y;
59 }
60 }
61
62 /// @brief 获取顶点
63 std::span<const SDL_Vertex> getVertices() const {
64 return m_vertices;
65 }
66
67 /// @brief 获取索引数组
68 std::span<const i32> getIndices() const {
69 return m_indices;
70 }
71
72 private:
73 std::array<SDL_Vertex, N> m_vertices; ///< 顶点
74 mutable std::array<i32, (N - 2) * 3> m_indices; ///< 索引
75 };
