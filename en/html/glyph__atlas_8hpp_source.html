26 /// @param ttfPath 字体文件
27 /// @param glyphSize 字号大小
28 /// @param atlasSize 字形图集大小
29 GlyphAtlas(const std::string_view &ttfPath, const float32 glyphSize, const int32 atlasSize)
30 : m_atlasSize(atlasSize), m_font(TTF_OpenFont(ttfPath.data(), glyphSize)),
31 m_atlasSurface(SDL_CreateSurface(atlasSize, atlasSize, SDL_PIXELFORMAT_RGBA32)) {
32 if(!m_font) {
33 const std::string errorMsg(ttfPath);
34 DAO_ERROR_LOG("字体文件加载失败:"+errorMsg);
35 }

39 SDL_GetPixelFormatDetails(m_atlasSurface->format),
40 nullptr, 0, 0, 0, 0

42 );

45 /// @brief 加载字形
46 /// @details 将一个文字的字形编码添加到字形图集中
47 /// @param charCode 文字的utf-32编码
48 void registerGlyph(const char32_t charCode) {
49 if (m_glyphs.contains(charCode)) { return; }
50 m_isUpdated = true;
51 SDL_Surface *glyphSurface = TTF_RenderGlyph_Blended(m_font, charCode, SDL_Color{255, 255, 255, 255});
52 m_cursor.rowHeight = std::max(m_cursor.rowHeight, glyphSurface->h);
53 if (m_cursor.x + glyphSurface->w > m_atlasSize) {
54 m_cursor.x = 0;
55 m_cursor.y += m_cursor.rowHeight;
56 m_cursor.rowHeight = 0;
57 }
58 const SDL_Rect dst{m_cursor.x, m_cursor.y, glyphSurface->w, glyphSurface->h};
59 SDL_BlitSurface(glyphSurface, nullptr, m_atlasSurface, &dst);
60 int minx, maxx, miny, maxy, advance;
61 TTF_GetGlyphMetrics(m_font, charCode, &minx, &maxx, &miny, &maxy, &advance);
62
63 Glyph glyph{};
64 glyph.size = {static_cast<float>(glyphSurface->w), static_cast<float>(glyphSurface->h)};
65 glyph.bearing = {static_cast<float>(minx), static_cast<float>(maxy)};
66 glyph.advance = static_cast<float>(advance);
67
68 glyph.pos = {
69 ratio(dst.x, m_atlasSize), ratio(dst.y, m_atlasSize),
70 ratio(dst.w, m_atlasSize), ratio(dst.h, m_atlasSize)
71 };
72
73 m_glyphs.emplace(charCode, glyph);
74 m_cursor.x += glyphSurface->w;
75
76 SDL_DestroySurface(glyphSurface);
77 }

79 /// @brief 批量加载字形
80 /// @details 将一批文字的字形编码添加到字形图集中
81 void batchRegisterGlyph(const std::string_view chars) {
82 for (auto &ch: chars) {
83 registerGlyph(ch);
84 }

88 BoundingBox getGlyphAtlasRegion(const char32_t charCode) {

93 };
